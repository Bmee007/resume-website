<!-- IntersectionObserver script contract (add once in BaseLayout body) -->
```js
// Fires once per element, unobserves after trigger
const observer = new IntersectionObserver(
  (entries) => {
    entries.forEach((entry) => {
      if (entry.isIntersecting) {
        entry.target.classList.add('is-visible');
        observer.unobserve(entry.target);
      }
    });
  },
  { rootMargin: '-10% 0px -10% 0px', threshold: 0 }
);
document.querySelectorAll('.section-animate').forEach((el) => observer.observe(el));
```

<!-- Hero CSS keyframe contract (scoped to HeroSection, no JS) -->
```css
@keyframes hero-fade-in {
  from { opacity: 0; transform: translateY(20px); }
  to   { opacity: 1; transform: translateY(0); }
}
.hero-heading { animation: hero-fade-in 600ms ease forwards; }
.hero-sub { opacity: 0; animation: hero-fade-in 600ms ease 150ms forwards; }
@media (prefers-reduced-motion: reduce) {
  .hero-heading, .hero-sub { animation: none; opacity: 1; transform: none; }
}
```
