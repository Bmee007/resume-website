```typescript
import { test, expect } from "@playwright/test";

test("CONT-01: contact section exists at bottom of page", async ({ page }) => {
  await page.goto("/");
  const contact = page.locator("#contact");
  await expect(contact).toBeVisible();
});

test("CONT-02: contact form has exactly 3 fields (name, email, message)", async ({ page }) => {
  await page.goto("/");
  const form = page.locator("#contact form");
  await expect(form.locator('input[name="name"]')).toBeVisible();
  await expect(form.locator('input[name="email"]')).toBeVisible();
  await expect(form.locator('textarea[name="message"]')).toBeVisible();
  // Confirm no 4th field exists
  const allInputs = form.locator("input, textarea, select");
  await expect(allInputs).toHaveCount(3);
});

test("CONT-03: form targets Formspree endpoint", async ({ page }) => {
  await page.goto("/");
  // Verify the form has a data-formspree-id attribute OR the page source contains formspree.io
  // Check data attribute on the form element
  const form = page.locator("#contact form");
  await expect(form).toBeVisible();
  const formId = await form.getAttribute("data-formspree-id");
  expect(formId).toBeTruthy();
});

test("CONT-04: error state shown when form submitted empty", async ({ page }) => {
  await page.goto("/");
  // Submit the form with all fields empty — client validation fires before any network call
  const submitBtn = page.locator("#contact form button[type='submit']");
  await submitBtn.click();
  // At least one field should have the is-invalid class
  const invalidField = page.locator("#contact .is-invalid").first();
  await expect(invalidField).toBeVisible();
});

test("CONT-05: LinkedIn profile link in contact section", async ({ page }) => {
  await page.goto("/");
  const linkedinLink = page.locator("#contact a[href*='linkedin.com/in/borinakeo']");
  await expect(linkedinLink.first()).toBeVisible();
});

test("CONT-06: GitHub profile link in contact section", async ({ page }) => {
  await page.goto("/");
  const githubLink = page.locator("#contact a[href*='github.com/Bmee007']");
  await expect(githubLink.first()).toBeVisible();
});
```

Important notes:
- CONT-03 checks `data-formspree-id` attribute on the form element — the implementation plan (04-02) must add this attribute.
- CONT-04 tests the error path only (empty submit triggers client validation without a network call). The success path (actual email delivery) is manual-only per VALIDATION.md.
- All tests scope to `#contact` to prevent strict mode violations.
- Do NOT add `--reporter=line` or watch flags.
