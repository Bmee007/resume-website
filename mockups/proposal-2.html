from openai import AzureOpenAI
import dynamics365 as d365

# AI-powered freight document classifier
client = AzureOpenAI(
  azure_endpoint="https://bk-openai.openai.azure.com",
  api_version="2024-05-01-preview"
)

def classify_and_route(freight_doc:str) -> dict:
  response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role":"user", "content": freight_doc}]
  )
  result = response.choices[0].message.content
  return d365.route_to_wms(result)
